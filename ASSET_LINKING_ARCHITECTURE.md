### Monitoring Queries
```sql
-- Total assets
SELECT COUNT(*) FROM assets;

-- Assets with links
SELECT COUNT(*) FROM assets 
WHERE linked_service_ids != '[]';

-- Most linked assets
SELECT asset_name, 
       jsonb_array_length(linked_service_ids::jsonb) as link_count
FROM assets 
ORDER BY link_count DESC 
LIMIT 10;

-- Assets by repository
SELECT tags as repository, COUNT(*) 
FROM assets 
GROUP BY tags;
```
